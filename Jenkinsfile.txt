pipeline {
    agent any
    stages {
        stage('---clean---') {
            steps {
                bat "rmdir /s /q JenkinsPipelinethree "
                bat "git clone https://github.com/sriramede9/JenkinsPipelinethree.git"
               // bat "cd JenkinsPipelinethree"
                bat "mvn clean -f JenkinsPipelinethree"
            }
        }
        stage('--test--') {
            steps {
                bat "mvn test -f JenkinsPipelinethree"
            }
        }
        stage('--package--') {
            steps {
                bat "mvn package -f JenkinsPipelinethree"
            }
        }
    }
}
